169Edge<T> * Vertex<T>::addEdgeChecked(Vertex<T> *d, double w) {
170 if (this == d) return nullptr; // Don't add edge to itself

172 // Check if edge already exists
173 for (Edge<T>* edge : adj) {
174 if (*edge->getDest()->getInfo() == *d->getInfo()) {
175 return nullptr;
176 }

185/*
186 * Auxiliary function to remove an outgoing edge (with a given destination (d))
187 * from a vertex (this).
188 * Returns true if successful, and false if such edge does not exist.
189 */
190template <class T>
191bool Vertex<T>::removeEdge(T in) {
192 bool removedEdge = false;
193 auto it = adj.begin();
194 while (it != adj.end()) {
195 Edge<T> *edge = *it;
196 Vertex<T> *dest = edge->getDest();
197 if (dest->getInfo() == in) {
198 it = adj.erase(it);
199 deleteEdge(edge);
200 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
201 }
202 else {
203 it++;
204 }
205 }
206 return removedEdge;
207}

409/*
410 * Removes a vertex with a given content (in) from a graph (this), and
411 * all outgoing and incoming edges.
412 * Returns true if successful, and false if such vertex does not exist.
413 */
414template <class T>
415bool Graph<T>::removeVertex(const T &in) {
416 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
417 if ((*it)->getInfo() == in) {
418 auto v = *it;
419 v->removeOutgoingEdges();
420 for (auto u : vertexSet) {
421 u->removeEdge(v->getInfo());
422 }
423 vertexSet.erase(it);
424 delete v;
425 return true;
426 }
427 }
428 return false;

482/*
483 * Performs a depth-first search (dfs) traversal in a graph (this).
484 * Returns a vector with the contents of the vertices by dfs order.
485 */
486template <class T>
487std::vector<T> Graph<T>::dfs() const {
488 std::vector<T> res;
489 for (auto v : vertexSet)
490 v->setVisited(false);
491 for (auto v : vertexSet)
492 if (!v->isVisited())
493 dfsVisit(v, res);
494 return res;
495}

497/*
498 * Performs a depth-first search (dfs) in a graph (this) from the source node.
499 * Returns a vector with the contents of the vertices by dfs order.
500 */
501template <class T>
502std::vector<T> Graph<T>::dfs(const T & source) const {
503 std::vector<int> res;
504 // Get the source vertex
505 auto s = findVertex(source);
506 if (s == nullptr) {
507 return res;
508 }
509 // Set that no vertex has been visited yet
510 for (auto v : vertexSet) {
511 v->setVisited(false);
512 }
513 // Perform the actual DFS using recursion
514 dfsVisit(s, res);
515
516 return res;
517}

519/*
520 * Auxiliary function that visits a vertex (v) and its adjacent, recursively.
521 * Updates a parameter with the list of visited node contents.
522 */
523template <class T>
524void Graph<T>::dfsVisit(Vertex<T> *v, std::vector<T> & res) const {
525 v->setVisited(true);
526 res.push_back(v->getInfo());
527 for (auto & e : v->getAdj()) {
528 auto w = e->getDest();
529 if (!w->isVisited()) {
530 dfsVisit(w, res);
531 }
532 }
533}

535/****************** BFS ********************/
536/*
537 * Performs a breadth-first search (bfs) in a graph (this), starting
538 * from the vertex with the given source contents (source).
539 * Returns a vector with the contents of the vertices by bfs order.
540 */
541template <class T>
542std::vector<T> Graph<T>::bfs(const T & source) const {
543 std::vector<int> res;
544 // Get the source vertex
545 auto s = findVertex(source);
546 if (s == nullptr) {
547 return res;
548 }
549
550 // Set that no vertex has been visited yet
551 for (auto v : vertexSet) {
552 v->setVisited(false);
553 }
554
555 // Perform the actual BFS using a queue
556 std::queue<Vertex<T> *> q;
557 q.push(s);
558 s->setVisited(true);
559 while (!q.empty()) {
560 auto v = q.front();
561 q.pop();
562 res.push_back(v->getInfo());
563 for (auto & e : v->getAdj()) {
564 auto w = e->getDest();
565 if ( ! w->isVisited()) {
566 q.push(w);
567 w->setVisited(true);
568 }
569 }
570 }
571 return res;
572}

576/****************** isDAG ********************/
577/*
578 * Performs a depth-first search in a graph (this), to determine if the graph
579 * is acyclic (acyclic directed graph or DAG).
580 * During the search, a cycle is found if an edge connects to a vertex
581 * that is being processed in the stack of recursive calls (see theoretical classes).
582 * Returns true if the graph is acyclic, and false otherwise.
583 */
584
585template <class T>
586bool Graph<T>::isDAG() const {
587 for (auto v : vertexSet) {
588 v->setVisited(false);
589 v->setProcesssing(false);
590 }
591 for (auto v : vertexSet) {
592 if (! v->isVisited()) {
593 if ( ! dfsIsDAG(v) ) return false;
594 }
595 }
596 return true;
597}
598
603template <class T>
604bool Graph<T>::dfsIsDAG(Vertex<T> *v) const {
605 v->setVisited(true);
606 v->setProcesssing(true);
607 for (auto e : v->getAdj()) {
608 auto w = e->getDest();
609 if (w->isProcessing()) return false;
610 if (! w->isVisited()) {
611 if (! dfsIsDAG(w)) return false;
612 }
613 }
614 v->setProcesssing(false);
615 return true;
616}

618/****************** toposort ********************/
619template<class T>
620std::vector<T> Graph<T>::topsort() const {
621 std::vector<int> res;
622
623 for (auto v : vertexSet) {
624 v->setIndegree(0);
625 }
626 for (auto v : vertexSet) {
627 for (auto e : v->getAdj()) {
628 unsigned int indegree = e->getDest()->getIndegree();
629 e->getDest()->setIndegree(indegree + 1);
630 }
631 }
632
633 std::queue<Vertex<T> *> q;
634 for (auto v : vertexSet) {
635 if (v->getIndegree() == 0) {
636 q.push(v);
637 }
638 }
639
640 while( !q.empty() ) {
641 Vertex<T> * v = q.front();
642 q.pop();
643 res.push_back(v->getInfo());
644 for(auto e : v->getAdj()) {
645 auto w = e->getDest();
646 w->setIndegree(w->getIndegree() - 1);
647 if(w->getIndegree() == 0) {
648 q.push(w);
649 }
650 }
651 }
652
653 if ( res.size() != vertexSet.size() ) {
654 //std::cout << "Impossible topological ordering!" << std::endl;
655 res.clear();
656 return res;
657 }
658
659 return res;
660}
